########################################################################################################################
shr.address
########################################################################################################################

========================================================================================================================
DATA ELEMENTS
========================================================================================================================

------------------------------------------------------------------------------------------------------------------------
Address         composition
------------------------------------------------------------------------------------------------------------------------
title           "Address"
description     "An address expressed using postal conventions (as opposed to GPS or other location definition formats).
                This data type may be used to convey addresses for use in delivering mail as well as for visiting
                locations and which might not be valid for mail delivery. There are a variety of postal address formats
                defined around the world. (Source: HL7 FHIR)"
contains        many        AddressLine
                optional    City
                optional    District
                optional    State
                optional    PostalCode
                optional    Country
                optional    AddressType
                optional    AddressUse
                optional    AddressPeriod

------------------------------------------------------------------------------------------------------------------------
AddressLine     text
------------------------------------------------------------------------------------------------------------------------
title           "Line"
description     "This component contains the house number, apartment number, street name, street direction, P.O. Box
                number, delivery hints, and similar address information. (Source: HL7 FHIR)"
sources         "HL7 FHIR: Address.line"

------------------------------------------------------------------------------------------------------------------------
City            text
------------------------------------------------------------------------------------------------------------------------
title           "City"
description     "The name of the city, town, village or other community or delivery center. (Source: HL7 FHIR)"
sources         "HL7 FHIR: Address.city"

------------------------------------------------------------------------------------------------------------------------
District        text
------------------------------------------------------------------------------------------------------------------------
title           "District (i.e. County)"
description     "The name of the administrative area (county). (Source: HL7 FHIR)"
sources         "HL7 FHIR: Address.district"

------------------------------------------------------------------------------------------------------------------------
State           text
------------------------------------------------------------------------------------------------------------------------
title           "State"
description     "Sub-unit of a country with limited sovereignty in a federally organized country. A code may be used if
                codes are in common use (i.e. US 2 letter state codes). (Source: HL7 FHIR)"
sources         "HL7 FHIR: Address.state"

------------------------------------------------------------------------------------------------------------------------
PostalCode      text
------------------------------------------------------------------------------------------------------------------------
title           "Postal Code"
description     "A postal code designating a region defined by the postal service. (Source: HL7 FHIR)"
sources         "HL7 FHIR: Address.postalCode"

------------------------------------------------------------------------------------------------------------------------
Country         text
------------------------------------------------------------------------------------------------------------------------
title           "Country"
description     "Country - a nation as commonly understood or generally accepted. (Source: HL7 FHIR)"
sources         "HL7 FHIR: Address.country"

------------------------------------------------------------------------------------------------------------------------
AddressType     enum
------------------------------------------------------------------------------------------------------------------------
title           "Address Type"
description     "Distinguishes between physical addresses (those you can visit) and mailing addresses (e.g. PO Boxes and
                care-of addresses). Most addresses are both. (Source: HL7 FHIR)"
values          #postal     displayed as "Postal"
                #physical   displayed as "Physical"
                #both       displayed as "Postal & Physical"
                // TODO: Should we support "both" or make this multi-value?

------------------------------------------------------------------------------------------------------------------------
AddressUse      enum
------------------------------------------------------------------------------------------------------------------------
title           "Address Use"
description     "The purpose of this address. (Source: HL7 FHIR)"
values          #home   displayed as "Home"
                #work   displayed as "Work"
                #temp   displayed as "Temporary"
                #old    displayed as "Old / Incorrect"

------------------------------------------------------------------------------------------------------------------------
AddressPeriod   period
------------------------------------------------------------------------------------------------------------------------
title           "Address Period"
description     "Time period when address was/is in use. (Source: HL7 FHIR)"
sources         "HL7 FHIR: Address.period"
