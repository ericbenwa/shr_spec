//----------------  Base Entry  -----------------------
// The base for all entries in the SHR

Group:			Entry
Description:	"The base for all entries in the SHR. Cpntains elements that describe the identification, timing, attribution, and other circumstances involved in an SHR entry"
1..1    		EntryID  
1..1			SubjectID
1..*			Concept
0..1			DateOfActivity
0..1    		DateOfDocumentation
1..1    		DateOfEntry
0..1    		InformationSource   
0..1			InformationDocumenter  // e.g. scribe or coder
0..1			InformationEnterer   // who actually input or created the entry  
0..1			Language    // language used in this entry (assumed to be EN unless otherwise stated)
0..1			Summary  	// text summary of this entry (from core)
1..1			Version     // Version of this entry (from core)
0..1			LastUpdated // TODO: If version are not stored, then we will need a log of all changes

	Element:    	EntryID
	Concept:        MTH#C0600091
	Description:    "A unique, persistent, permanent identifier for an entry in a health record"
	Value:          id
	
	Element: 		SubjectID
	Concept:		TBD
	Description:	"The persistent, unique ID of the person described by and tracked by this record. This ID has no intrinsic meaning and should not contain any identifiable information, such a mashup of name and birth date."
	Value:			id
	
	Element:		Concept
	Concept:		TBD  // the concept of a concept
	Description:	"The (sub)concept identifying the question, observation, measurement, or action. Multiple codings are allowed. Can include translations, synonyms and hypernyms. For example, if the concept is sitting blood pressure, the hypernyms blood pressure and vital sign can be part of the coding."
	Value:			Coding
	
	Element:		DateOfActivity
	Concept:		TBD
	Description:	"The date or date range when the documented activity (observation, assessment) took place"
	Value:			(dateTime or date or Period)
	
	Element:    	DateOfDocumentation
	Concept:        MTH#C2347859
	Description:    "The date on which content was created, or content creation began"
	Value:          (dateTime or date or Period)

	Element:   	 	DateOfEntry
	Concept:        MTH#C3669169
	Description:    "The point in time when the information was recorded in the system of record."
	Value:          dateTime
	
	Group:    		InformationSource
	Concept:        MTH#C0449416
	Description:    "From whom or what the information originates, as distinct from the actor creating the entry, e.g. the subject (patient), medical professional, family member, device or software program."
	1..1	        CareProfessional or FamilyMember or UnrelatedThirdParty // or ComputerProgram or Device or Organization

	Element:		InformationDocumenter
	Concept:		TBD
	Description:	"The person, device, or computer program that documented the activity, observation, or assessment."
	Value:			HealthcareActor // or ComputerProgram or Device 

	Element:		InformationEnterer
	Concept:		TBD
	Description:	"The person, device, or computer program that actually entered the information into the system of record."
	Value:			HealthcareActor // or ComputerProgram or Device 
	
	Element:    	RecordedBy
	Concept:        MTH#C1707531
	Description:    "Indicates the person or organization who brought the item into existence"
	Value:          HealthcareActor	
	
	Element:		Language
	Concept:		MTH#C0023008
	Description: 	"An identifier for a human language"
	Value:			code from http://hl7.org/fhir/ValueSet/languages

	Element: 		LastUpdated
	Concept:		TBD
	Description:	"The time the entry was last updated (changed)"
	Value:			dateTime
